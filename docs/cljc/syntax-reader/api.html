(defn read-tags
  ([n tags]
   (read-tags n tags {}))

  ([n tags options]
   (letfn [(f0 [result state {:keys [ending-tag] :as metafunctions}]
               (if-let [ending-tag (ending-tag)]
                       (read.utils/read-found-tag result state metafunctions ending-tag)
                       (-> result)))]
          (interpreter/interpreter n f0 {} tags options))))

   (remove-tags n tags {}))

  ([n tags options]

(defn tag-starting-position
  ([n searched-tag]
   (tag-starting-position n searched-tag {} {}))

  ([n searched-tag tags]
   (tag-starting-position n searched-tag tags {}))

  ([n [tag-name & _ :as searched-tag] tags options]
   (letfn [
           (f0 [_ {:keys [cursor] :as state} {:keys [stop tag-starts?] :as metafunctions}]
               (if (tag-starts? tag-name)
                   (stop cursor)))]
          (let [tags (core.prototypes/tags-prototype tags)
                tags (vector/cons-item tags searched-tag)]
               (syntax-interpreter/interpreter n f0 nil tags options)))))

(defn update-tags
  ([n tags]
   (update-tags n tags {}))

  ([n tags options]
