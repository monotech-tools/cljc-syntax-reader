(defn read-tags
  ([n tags]
   (read-tags n tags {}))

  ([n tags options]
   (letfn [(f0 [result state {:keys [ending-tag] :as metafunctions}]
               (if-let [ending-tag (ending-tag)]
                       (read.utils/read-found-tag result state metafunctions ending-tag)
                       (-> result)))]
          (interpreter/interpreter n f0 {} tags options))))
